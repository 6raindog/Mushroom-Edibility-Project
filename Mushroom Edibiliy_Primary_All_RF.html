from sklearn.preprocessing import LabelEncoder
from sklearn.preprocessing import OneHotEncoder
from sklearn.preprocessing import MinMaxScaler

# choose dataset
dataset = primary_dataset

# label encode y
target_encoder = LabelEncoder()

target_encoded = target_encoder.fit_transform(dataset.target)
# p = 1, e = 0

# one hot encode x
X = dataset.drop(["target"], axis=1)
feature_encoder = OneHotEncoder(sparse_output=False, feature_name_combiner='concat')
X_encoded = feature_encoder.fit_transform(X)

if dataset is secondary_dataset or dataset is combined_dataset:
    continuous_data = dataset[["cap diameter","stalk height","stalk width"]]

    # handle continuous data (secondary and combined only)
    scaler = MinMaxScaler()
    scaled_data = scaler.fit_transform(continuous_data)

    # put scaled data back into main feature array
    X_encoded = np.column_stack((X_encoded, scaled_data))
    print("Continuous variables dealt with")
